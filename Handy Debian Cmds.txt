----Reboot---
sudo reboot

----get available updates from repository and apply the updates---
sudo apt-get update		//update repositories
sudo apt-get upgrade	//apply updates

----Configure Static IP---
sudo nano /etc/dhcpcd.conf		//reboot to take effect

interface eth0
static ip_address=192.168.0.10/24
static routers=192.168.0.1
static domain_name_servers=192.168.0.1

interface wlan0
static ip_address=192.168.0.200/24
static routers=192.168.0.1
static domain_name_servers=192.168.0.1

----Install pi-hole---
curl -sSL https://install.pi-hole.net | bash

----Remove Folder Recursively---
sudo rm -r $FolderName

----Check disk is HDD or SSD---
cat /sys/block/sda/queue/rotational		//returns 1 for HDD, 0 for SSD

----change ownership of a partition to yourself---
sudo chown -R -v $USER: /media/$USERNAME/"$partitionname"

----Give full read/write/execute access to yourself on a partition---
sudo chmod 755 /media/$USERNAME/"$partitionname"			/-r for recursive, -v for verbose, 775 for yourself

----display crontab contents---
crontab -l

----edit cron jobs---
crontab -e

----remove all cron jobs---
$ crontab -r

----show just cron jobs in log file---
grep CRON /var/log/syslog		//default location of cron jobs log and many other -> var/log/syslog


----Update OS---
sudo sh -c "apt update && apt dist-upgrade && apt autoremove"
